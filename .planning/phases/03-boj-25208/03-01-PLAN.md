    CONTEXT.md 설계 기반 solve() 완전 구현:
    - 면 번호: 0=뚫린면(초기바닥), 1=윗면, 2=북, 3=남, 4=동, 5=서
    - 초기 큐브 배열: faces[6] = {0,1,2,3,4,5} (인덱스=포지션: 0바닥,1윗,2북,3남,4동,5서)
    - s = north_face*6 + bottom_face (범위 0~35)
    - 초기 s = 2*6+0 = 12
    - 전환 테이블: buildRollTable()로 초기 faces 배열 BFS → 24상태×4방향 자동 생성
    - 굴리기 순열 (4방향):
      - 북: 바닥↔북, 윗면↔남 (동서 불변)
      - 남: 바닥↔남, 윗면↔북 (동서 불변)
      - 동: 바닥↔동, 윗면↔서 (남북 불변)
      - 서: 바닥↔서, 윗면↔동 (남북 불변)
    - BFS: dp[r][c][s] = 최소이동횟수, INF=math.MaxInt32
    - R칸 도달 시: bottom_face==0이면 승리 반환, 아니면 스킵
    - 전체 탐색 후 R 도달 불가 → -1

    ```go
    package main

    import (
    	"bufio"
    	"container/list"
    	"fmt"
    	"math"
    	"os"
    )

    var reader *bufio.Reader
    var writer *bufio.Writer

    // posFaces: [바닥=0, 윗=1, 북=2, 남=3, 동=4, 서=5] 포지션에 있는 면 번호
    type cubeState [6]int

    // rollTable[s][dir] = 다음 s (dir: 0=북,1=남,2=동,3=서)
    var rollTable [36][4]int

    // rollDir: 방향별 (바닥,윗,북,남,동,서) 포지션 순열
    // 북으로 굴리기: 바닥←북, 북←윗, 윗←남, 남←바닥 (동서 불변)
    func applyRoll(f cubeState, dir int) cubeState {
    	n := f
    	switch dir {
    	case 0: // 북
    		n[0], n[2], n[1], n[3] = f[2], f[1], f[3], f[0]
    	case 1: // 남
    		n[0], n[3], n[1], n[2] = f[3], f[1], f[2], f[0]
    	case 2: // 동
    		n[0], n[4], n[1], n[5] = f[4], f[1], f[5], f[0]
    	case 3: // 서
    		n[0], n[5], n[1], n[4] = f[5], f[1], f[4], f[0]
    	}
    	return n
    }

    func stateID(f cubeState) int {
    	return f[2]*6 + f[0] // north_face*6 + bottom_face
    }

    func buildRollTable() {
    	init0 := cubeState{0, 1, 2, 3, 4, 5}
    	visited := make(map[int]cubeState)
    	queue := list.New()
    	queue.PushBack(init0)
    	visited[stateID(init0)] = init0
    	for queue.Len() > 0 {
    		cur := queue.Remove(queue.Front()).(cubeState)
    		sid := stateID(cur)
    		for d := 0; d < 4; d++ {
    			nxt := applyRoll(cur, d)
    			nid := stateID(nxt)
    			rollTable[sid][d] = nid
    			if _, ok := visited[nid]; !ok {
    				visited[nid] = nxt
    				queue.PushBack(nxt)
    			}
    		}
    	}
    }

    var dr4 = [4]int{-1, 1, 0, 0}
    var dc4 = [4]int{0, 0, 1, -1}

    func solve(grid []string, dr, dc, rr, rc int) int {
    	n := len(grid)
    	m := len(grid[0])
    	const inf = math.MaxInt32
    	dp := make([][][36]int, n)
    	for i := range dp {
    		dp[i] = make([][36]int, m)
    		for j := range dp[i] {
    			for s := range dp[i][j] {
    				dp[i][j][s] = inf
    			}
    		}
    	}
    	initS := 12 // bottom=0, north=2
    	dp[dr][dc][initS] = 0
    	type state struct{ r, c, s int }
    	queue := list.New()
    	queue.PushBack(state{dr, dc, initS})
    	for queue.Len() > 0 {
    		cur := queue.Remove(queue.Front()).(state)
    		cost := dp[cur.r][cur.c][cur.s]
    		for d := 0; d < 4; d++ {
    			nr, nc := cur.r+dr4[d], cur.c+dc4[d]
    			if nr < 0 || nr >= n || nc < 0 || nc >= m {
    				continue
    			}
    			if grid[nr][nc] == '#' {
    				continue
    			}
    			ns := rollTable[cur.s][d]
    			bottom := ns % 6
    			if nr == rr && nc == rc {
    				if bottom == 0 {
    					return cost + 1
    				}
    				continue // 막힌 면 — 스킵
    			}
    			if cost+1 < dp[nr][nc][ns] {
    				dp[nr][nc][ns] = cost + 1
    				queue.PushBack(state{nr, nc, ns})
    			}
    		}
    	}
    	return -1
    }

    func main() {
    	reader = bufio.NewReader(os.Stdin)
    	writer = bufio.NewWriter(os.Stdout)
    	defer writer.Flush()
    	buildRollTable()
    	var n, m int
    	fmt.Fscan(reader, &n, &m)
    	grid := make([]string, n)
    	var startR, startC, robR, robC int
    	for i := 0; i < n; i++ {
    		fmt.Fscan(reader, &grid[i])
    		for j, ch := range grid[i] {
    			switch ch {
    			case 'D':
    				startR, startC = i, j
    			case 'R':
    				robR, robC = i, j
    			}
    		}
    	}
    	fmt.Fprintln(writer, solve(grid, startR, startC, robR, robC))
    }
    ```

    - 문제 링크: https://www.acmicpc.net/problem/25208
    - 목표: 탐정이 한 면이 뚫린 감옥을 굴려 도둑 위치에 뚫린 면이 바닥으로 도달
